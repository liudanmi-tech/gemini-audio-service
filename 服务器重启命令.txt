# 在服务器终端执行以下命令重启服务

cd ~/gemini-audio-service

# 停止旧服务
pkill -f "python.*main.py"
sleep 2

# 启动服务
source venv/bin/activate
nohup python3 main.py > ~/gemini-audio-service.log 2>&1 &
sleep 5

# 检查服务状态
ps aux | grep "[p]ython.*main.py"

# 查看日志
tail -20 ~/gemini-audio-service.log | grep -E "启动|Uvicorn|Application startup|ERROR"

# 测试路由（应该返回认证错误，而不是422）
curl -s http://localhost:8001/api/v1/profiles | head -5
